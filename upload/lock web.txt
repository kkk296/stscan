  // --- Gann Logic ---
  const GANN_STEPS = {
    buy: { up: 0.333, avg: 0.250, sl: 0.166, t1: 0.416, t2: 0.500, t3: 0.666 },
    sell: { below: -0.333, avg: -0.250, sl: 0.166, t1: -0.416, t2: -0.500, t3: -0.666 }
  };

  function calculateGann(price, multiplier) {
    const baseSqrt = Math.sqrt(price);
    const step = (val) => Math.pow(baseSqrt + (val * multiplier), 2);
    
    return {
      buy: {
        entry: step(GANN_STEPS.buy.up),
        avg: step(GANN_STEPS.buy.avg),
        sl: step(GANN_STEPS.buy.sl),
        t1: step(GANN_STEPS.buy.t1),
        t2: step(GANN_STEPS.buy.t2),
        t3: step(GANN_STEPS.buy.t3)
      },
      sell: {

        avg: step(GANN_STEPS.sell.avg),

  // --- Data Fetching (Real-Time) ---
  async function fetchMarketData() {
    document.getElementById('loader').classList.add('active');
    
    let stocks = [];

    try {
      const proxyUrl = "https://api.allorigins.win/get?url=";
      
      const promises = tickers.map(async (t) => {
        try {
          const url = `${proxyUrl}${encodeURIComponent(`https://query1.finance.yahoo.com/v8/finance/chart/${t}?interval=1d&range=5d`)}`;
          const res = await fetch(url);
          const data = await res.json();
          const json = JSON.parse(data.contents);
          
          if(!json.chart || !json.chart.result) return null;

          const meta = json.chart.result[0].meta;
          const quotes = json.chart.result[0].indicators.quote[0];
          const timestamps = json.chart.result[0].timestamp;

          const livePrice = meta.regularMarketPrice;
          const todayIdx = timestamps.length - 1;
          const yestIdx = todayIdx - 1;
          const validYestIdx = yestIdx >= 0 ? yestIdx : 0;

          // Handle Indices formatting (^NSEI -> NIFTY 50)
          let symbolName = t.replace('.NS', '');
          if(symbolName === "^NSEI") symbolName = "NIFTY 50";
          else if(symbolName === "^NSEBANK") symbolName = "BANK NIFTY";
          else if(symbolName === "^NIFTYFIN") symbolName = "FINNIFTY";

          const prevOpen = quotes.open[validYestIdx];
          const prevHigh = quotes.high[validYestIdx];
          const prevLow = quotes.low[validYestIdx];
          const prevClose = quotes.close[validYestIdx];
          
          const change = livePrice - prevClose;
          const pChange = (change / prevClose) * 100;

          return {
            symbol: symbolName,
            price: livePrice,
            change: change,
            pChange: pChange,
            o: prevOpen,
            h: prevHigh,
            l: prevLow,
            c: prevClose
          };
        } catch (e) {
          return null;
        }
      });

      const results = await Promise.all(promises);
      stocks = results.filter(r => r !== null);

    } catch (err) {
      console.error("Fetch error", err);
      showPopup("System", "Using offline data mode (CORS Restricted)", "neutral");
      stocks = generateOfflineData(tickers);
    }

    appData.gainers = stocks.filter(s => s.pChange > 0).sort((a,b) => b.pChange - a.pChange);
    appData.losers = stocks.filter(s => s.pChange < 0).sort((a,b) => a.pChange - b.pChange);
    
    const firstStock = appData.gainers[0] || appData.losers[0];
    if(firstStock) selectStock(firstStock.symbol);
    
    renderList('buy');
    document.getElementById('loader').classList.remove('active');
  }

  function generateOfflineData(tickers) {
    return tickers.map(t => {
      let symbolName = t.replace('.NS', '');
      if(symbolName === "^NSEI") symbolName = "NIFTY 50";
      else if(symbolName === "^NSEBANK") symbolName = "BANK NIFTY";
      else if(symbolName === "^NIFTYFIN") symbolName = "FINNIFTY";

      const basePrice = Math.random() * 2000 + 100; 
      const changeP = (Math.random() * 4) - 2; 
      const livePrice = basePrice * (1 + changeP/100);
      return {
        symbol: symbolName,
        price: livePrice,
        change: livePrice - basePrice,
        pChange: changeP,
        o: basePrice * 0.99,
        h: basePrice * 1.02,
        l: basePrice * 0.98,
        c: basePrice
      };
    });
  }

  // --- Alert Monitor ---
  let lastAlertTime = 0;
  function monitorAlerts() {
    if(!appData.currentStock) return;

    const alertInput = document.getElementById('inpAlert');
    const alertPrice = parseFloat(alertInput.value);
    const livePrice = appData.currentStock.price;

    if(alertPrice && livePrice) {
      const now = Date.now();
      if(now - lastAlertTime > 10000) {
        
        if(livePrice >= alertPrice) {
          alertInput.classList.remove('alert-active-sell');
          alertInput.classList.add('alert-active-buy');
          // Speak the alert
          speakAlert(`Alert! ${appData.currentStock.symbol} crossed ${alertPrice}`);
          showPopup("🚀 BREAKOUT ALERT", `${appData.currentStock.symbol} crossed ${alertPrice}!`, "alert");
          lastAlertTime = now;
        } else if (livePrice <= alertPrice) {
           alertInput.classList.remove('alert-active-buy');
           alertInput.classList.add('alert-active-sell');
           // Speak the alert
           speakAlert(`Alert! ${appData.currentStock.symbol} fell below ${alertPrice}`);
           showPopup("📉 BREAKDOWN ALERT", `${appData.currentStock.symbol} fell below ${alertPrice}!`, "alert");
           lastAlertTime = now;
        } else {
          alertInput.classList.remove('alert-active-buy', 'alert-active-sell');
        }
      }
    }
  }

  // --- Voice Alert Function (Replaces Beep) ---
  function speakAlert(text) {
    // Browser ke built-in Text-to-Speech use karein

  function switchTab(type) {
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    document.querySelector(`.tab[data-type="${type}"]`).classList.add('active');
    renderList(type);
  }

  function renderList(type) {
    const container = document.getElementById('stockListContainer');
    const list = type === 'buy' ? appData.gainers : appData.losers;
    
    let html = '';
    if(list.length === 0) html = '<div style="padding:20px; text-align:center; color:#666">No signals found</div>';
    
    list.forEach(s => {
      const colorClass = s.pChange >= 0 ? 'up' : 'down';
      const arrow = s.pChange >= 0 ? '▲' : '▼';
      html += `
        <div class="stock-item" onclick="selectStock('${s.symbol}')">
          <div>
            <span class="s-name">${s.symbol}</span>
            <span class="s-price">₹${s.price.toFixed(2)}</span>
          </div>
          <div class="s-change ${colorClass}">
            ${arrow} ${Math.abs(s.pChange).toFixed(2)}%
          </div>
        </div>
      `;
    });
    container.innerHTML = html;
  }

  function updateInputPrice() {
    if(!appData.currentStock) return;
    
    const baseType = document.getElementById('inpBase').value;
    let price = 0;

    if(baseType === 'prevClose') price = appData.currentStock.c;
    else if(baseType === 'prevHigh') price = appData.currentStock.h;
    else if(baseType === 'prevLow') price = appData.currentStock.l;
    else price = appData.currentStock.price;

    document.getElementById('inpPrice').value = price.toFixed(2);
    manualCalculate(); 
  }

  function selectStock(symbol) {
    const stock = [...appData.gainers, ...appData.losers].find(s => s.symbol === symbol);
    if(!stock) return;

    appData.currentStock = stock;

    document.getElementById('inpSymbol').value = stock.symbol;
    
    document.getElementById('ohlc-o').innerText = stock.o.toFixed(2);
    document.getElementById('ohlc-h').innerText = stock.h.toFixed(2);
    document.getElementById('ohlc-l').innerText = stock.l.toFixed(2);
    document.getElementById('ohlc-c').innerText = stock.c.toFixed(2);

    document.querySelectorAll('.stock-item').forEach(el => {
      el.classList.remove('selected');
      if(el.innerText.includes(symbol)) el.classList.add('selected');
    });

    if(stock.pChange > 1.5) showPopup("Buy Signal", `${stock.symbol} is strong!`, "buy");
    else if(stock.pChange < -1.5) showPopup("Sell Signal", `${stock.symbol} is weak!`, "sell");

    document.getElementById('inpAlert').value = '';
    document.getElementById('inpAlert').classList.remove('alert-active-buy', 'alert-active-sell');
    
    updateInputPrice();
  }

  function manualCalculate() {
    const price = parseFloat(document.getElementById('inpPrice').value);
    const mult = parseFloat(document.getElementById('inpMult').value);
    if(!price) return;

    const levels = calculateGann(price, mult);

    document.getElementById('buy-entry').innerText = levels.buy.entry.toFixed(2);
    document.getElementById('buy-avg').innerText = levels.buy.avg.toFixed(2);
    document.getElementById('buy-sl').innerText = levels.buy.sl.toFixed(2);
    document.getElementById('buy-t1').innerText = levels.buy.t1.toFixed(2);
    document.getElementById('buy-t2').innerText = levels.buy.t2.toFixed(2);
    document.getElementById('buy-t3').innerText = levels.buy.t3.toFixed(2);

    document.getElementById('sell-entry').innerText = levels.sell.entry.toFixed(2);
    document.getElementById('sell-avg').innerText = levels.sell.avg.toFixed(2);
    document.getElementById('sell-sl').innerText = levels.sell.sl.toFixed(2);
    document.getElementById('sell-t1').innerText = levels.sell.t1.toFixed(2);
    document.getElementById('sell-t2').innerText = levels.sell.t2.toFixed(2);
    document.getElementById('sell-t3').innerText = levels.sell.t3.toFixed(2);
    
    const term = document.getElementById('inpTerm').value;
  }

  function showPopup(title, msg, type) {
    const container = document.getElementById('popup-container');
    const el = document.createElement('div');
    el.className = `popup-card ${type}`;
    
    let icon = type === 'buy' ? '🟢' : (type === 'sell' ? '🔴' : 'ℹ️');
    
    el.innerHTML = `
      <div class="popup-title">${icon} ${title}</div>
      <div class="popup-msg">${msg}</div>
    `;
    
    container.appendChild(el);
    
    setTimeout(() => {
      el.style.opacity = '0';
      setTimeout(() => el.remove(), 300);
    }, 3000);
  }
